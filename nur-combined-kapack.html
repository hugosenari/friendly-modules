services.phpfpm0.pools

PHP-FPM pools. If no pools are defined, the PHP-FPM service is disabled.

type

attribute set of submodule

example

{
  services.phpfpm0.pools = {
    _type = "literalExpression";
    text = ''
      {
        mypool = {
          user = "php";
          group = "php";
          phpPackage = pkgs.php;
          settings = ''
            "pm" = "dynamic";
            "pm.max_children" = 75;
            "pm.start_servers" = 10;
            "pm.min_spare_servers" = 5;
            "pm.max_spare_servers" = 20;
            "pm.max_requests" = 500;
          '';
        }
      }
    '';
  };
}

default

{
  services.phpfpm0.pools = {};
}
