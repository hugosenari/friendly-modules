services.dyndnsc.profiles

Declarative profile config

type

attribute set of submodule

example

{
  services.dyndnsc.profiles = {
    _type = "literalExpression";
    text = ''
      {
        "myhost_ip4" = {
          preset = "nsupdate.info:ipv4";
          hostname = "dynamic.example.com";
          username = "me";
          passwordFile = "/run/keys/dyndnsc-myhost_ip4";
        };
      };
      
    '';
  };
}
